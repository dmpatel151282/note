
RTC 精确到分钟

<linux/rtc.h>
struct rtc_time {
    int tm_sec;
    int tm_min;
    int tm_hour;
    int tm_mday;
    int tm_mon;
    int tm_year;
    int tm_wday;
    int tm_yday;
    int tm_isdst;
};

RTC_ALM_READ              读取闹钟时间
RTC_ALM_SET               设置闹钟时间
arg: struct rtc_time *

RTC_RD_TIME               读取当前时间
RTC_SET_TIME              设置当前时间
arg: struct rtc_time *

RTC_AIE_ON                闹钟中断使能
read(fileDesc, &data, sizeof(unsigned long)); 阻塞，直到闹钟时间到了
RTC_AIE_OFF               闹钟中断关闭
arg: int 

RTC_UIE_ON                update中断使能
RTC_UIE_OFF               update中断关闭
arg: int
